40  void CPointMass3DDroneModel::Reset() {

42  m_cPosition = GetEmbodiedEntity().GetOriginAnchor().Position;
43  /* reset the home position (may have been changed by MoveTo or by added noise) */
44  m_cHomePosition = m_cPosition;
45  /* reset the drone's orientation */
46  CRadians cYaw, cPitch, cRoll;
47  GetEmbodiedEntity().GetOriginAnchor().Orientation.ToEulerAngles(cYaw, cPitch, cRoll);
48  m_cOrientation.SetX(cRoll.GetValue());
49  m_cOrientation.SetY(cPitch.GetValue());
50  m_cOrientation.SetZ(cYaw.GetValue());
51  /* reset the home yaw angle (may have been changed by MoveTo or by added noise) */
52  m_fHomeYawAngle = cYaw.GetValue();
53  /* reset the drone's velocity */
54  m_cVelocity.Set(0.0, 0.0, 0.0);
55  m_cVelocityPrev.Set(0.0, 0.0, 0.0);
56  /* reset the drone's angular velocity */
57  m_cAngularVelocity.Set(0.0, 0.0, 0.0);
58  m_cAngularVelocityPrev.Set(0.0, 0.0, 0.0);
59  /* reset the drone's previous acceleration */
60  m_cAccelerationPrev.Set(0.0, 0.0, 0.0);
61  /* reset the drone's previous angular acceleration */
62  m_cAngularAccelerationPrev.Set(0.0, 0.0, 0.0);
63  /* reset the variables for PID controller */
64  m_cOrientationTargetPrev.Set(0.0, 0.0, 0.0);
65  m_cAngularVelocityCumulativeError.Set(0.0, 0.0, 0.0);
66  m_fAltitudeCumulativeError = 0.0;
67  m_fTargetPositionZPrev = 0.0;
68  /* reset the gyro bias */
69  m_fGyroBias = MEMS_GYRO_BIAS_INIT;
70  m_fAccelBias = MEMS_ACCEL_BIAS_INIT;
71  m_fAngleRandomWalk = MEMS_GYRO_RANDOM_WALK_INIT;

112  void CPointMass3DDroneModel::UpdatePhysics() {
113  /* update the position (XY) and altitude (Z) controller */
114  CVector3 cPositionError(m_cInputPosition - m_cPosition);
115  /* accumulate the altitude error */
116  m_fAltitudeCumulativeError += cPositionError.GetZ() * GetPM3DEngine().GetPhysicsClockTick();
117  /* calculate the azimuth contribution to the navigation of drone on XY plane */
118  Real fAzimuth = std::atan2(std::abs(cPositionError.GetY()),
119  std::abs(cPositionError.GetX()));
120  /* calculate velocity limits */
121  CRange<Real> cVelocityLimitX(-XY_VEL_MAX * std::cos(fAzimuth), XY_VEL_MAX * std::cos(fAzimuth));
122  CRange<Real> cVelocityLimitY(-XY_VEL_MAX * std::sin(fAzimuth), XY_VEL_MAX * std::sin(fAzimuth));
123  CRange<Real> cVelocityLimitZ(-Z_VEL_MAX, Z_VEL_MAX);
124  /* calculate desired XYZ velocities */
125  Real fTargetTransVelX =
126  cPositionError.GetX() * XY_POS_KP;
127  Real fTargetTransVelY =
128  cPositionError.GetY() * XY_POS_KP;
129  Real fTargetTransVelZ =
130  (m_cInputPosition.GetZ() - m_fTargetPositionZPrev) / GetPM3DEngine().GetPhysicsClockTick();
131  /* saturate velocities */
132  cVelocityLimitX.TruncValue(fTargetTransVelX);
133  cVelocityLimitY.TruncValue(fTargetTransVelY);
134  cVelocityLimitZ.TruncValue(fTargetTransVelZ);
135  /* store the previous desired altitude for the altitude PID calculation */
136  m_fTargetPositionZPrev = m_cInputPosition.GetZ();
137  /* store XYZ velocity error */
138  CVector3 cTransVelocityError(fTargetTransVelX, fTargetTransVelY, fTargetTransVelZ);
139  cTransVelocityError -= m_cVelocity;
140  /* calculate XY desired accelerations */
141  CVector3 cTargetTransAcc = cTransVelocityError * XY_VEL_KP;
142  /* yaw angle correction to get downward Z axis in the right handed coordinate system */
143  Real fYawAngleCorrected(m_cOrientation.GetZ() + M_PI);
144  /* outputs of the position controller */
145  Real fDesiredRollAngle = std::cos(m_cOrientation.GetY()) * std::cos(m_cOrientation.GetX()) *
146  (std::sin(fYawAngleCorrected) * cTargetTransAcc.GetX() - std::cos(fYawAngleCorrected) * cTargetTransAcc.GetY()) / m_cPM3DEngine.GetGravity();
147  Real fDesiredPitchAngle = std::cos(m_cOrientation.GetY()) * std::cos(m_cOrientation.GetX()) *
148  (std::cos(fYawAngleCorrected) * cTargetTransAcc.GetX() + std::sin(fYawAngleCorrected) * cTargetTransAcc.GetY()) / m_cPM3DEngine.GetGravity();
149  Real fDesiredYawAngle = m_fInputYawAngle;
150  /* saturate the outputs of the position controller */
151  ROLL_PITCH_LIMIT.TruncValue(fDesiredRollAngle);
152  ROLL_PITCH_LIMIT.TruncValue(fDesiredPitchAngle);
153  /* store the desired orientation values in a vector */
154  CVector3 cOrientationTarget(fDesiredRollAngle, fDesiredPitchAngle, fDesiredYawAngle);
155  /* output of the altitude controller */
156  Real fAltitudeControlSignal = MASS * GetPM3DEngine().GetGravity() +
157  CalculatePIDResponse(cPositionError.GetZ(),
158  m_fAltitudeCumulativeError,
159  cTransVelocityError.GetZ(),
160  ALTITUDE_KP,
161  ALTITUDE_KI,
162  ALTITUDE_KD) / (std::cos(m_cOrientation.GetX()) * std::cos(m_cOrientation.GetY()));
163  /*** attitude (roll, pitch, yaw) control ***/
164  /* roll, pitch, yaw errors */
165  CVector3 cOrientationError(cOrientationTarget - m_cOrientation);
166  /* desired roll, pitch, yaw rates */
167  CVector3 cAngularVelocityTarget =
168  (cOrientationTarget - m_cOrientationTargetPrev) / GetPM3DEngine().GetPhysicsClockTick();
169  /* previous desired roll, pitch, yaw values for the controllers */
170  m_cOrientationTargetPrev = cOrientationTarget;
171  /* rotational rate errors */
172  CVector3 cAngularVelocityError(cAngularVelocityTarget - m_cAngularVelocity);
173  /* accumulated roll, pitch, yaw errors for the controllers */
174  m_cAngularVelocityCumulativeError +=
175  cAngularVelocityError * GetPM3DEngine().GetPhysicsClockTick();
176  /* roll controller output signal */
177  Real fAttitudeControlSignalX = INERTIA.GetX() *
178  CalculatePIDResponse(cOrientationError.GetX(),
179  m_cAngularVelocityCumulativeError.GetX(),
180  cAngularVelocityError.GetX(),
181  ROLL_PITCH_KP,
182  ROLL_PITCH_KI,
183  ROLL_PITCH_KD);
184  /* pitch controller output signal */
185  Real fAttitudeControlSignalY = INERTIA.GetY() *
186  CalculatePIDResponse(cOrientationError.GetY(),
187  m_cAngularVelocityCumulativeError.GetY(),
188  cAngularVelocityError.GetY(),
189  ROLL_PITCH_KP,
190  ROLL_PITCH_KI,
191  ROLL_PITCH_KD);
192  /* yaw controller output signal */
193  Real fAttitudeControlSignalZ = INERTIA.GetZ() *
194  CalculatePIDResponse(cOrientationError.GetZ(),
195  m_cAngularVelocityCumulativeError.GetZ(),
196  cAngularVelocityError.GetZ(),
197  YAW_KP,
198  YAW_KI,
199  YAW_KD);
200  /* saturate controller outputs */
201  TORQUE_LIMIT.TruncValue(fAttitudeControlSignalX);
202  TORQUE_LIMIT.TruncValue(fAttitudeControlSignalY);
203  TORQUE_LIMIT.TruncValue(fAttitudeControlSignalZ);
204  /* calculate the rotor speeds from the control signals */
205  std::array<Real, 4> arrSquaredRotorSpeeds = {
206  fAltitudeControlSignal / (4 * B) -
207  fAttitudeControlSignalX * (ROOT_TWO / (4 * B * ARM_LENGTH)) -
208  fAttitudeControlSignalY * (ROOT_TWO / (4 * B * ARM_LENGTH)) -
209  fAttitudeControlSignalZ / (4 * D),
210  fAltitudeControlSignal / (4 * B) -
211  fAttitudeControlSignalX * (ROOT_TWO / (4 * B * ARM_LENGTH)) +
212  fAttitudeControlSignalY * (ROOT_TWO / (4 * B * ARM_LENGTH)) +
213  fAttitudeControlSignalZ / (4 * D),
214  fAltitudeControlSignal / (4 * B) +
215  fAttitudeControlSignalX * (ROOT_TWO / (4 * B * ARM_LENGTH)) +
216  fAttitudeControlSignalY * (ROOT_TWO / (4 * B * ARM_LENGTH)) -
217  fAttitudeControlSignalZ / (4 * D),
218  fAltitudeControlSignal / (4 * B) +
219  fAttitudeControlSignalX * (ROOT_TWO / (4 * B * ARM_LENGTH)) -
220  fAttitudeControlSignalY * (ROOT_TWO / (4 * B * ARM_LENGTH)) +
221  fAttitudeControlSignalZ / (4 * D),
222  };
223  /* calculate the effect of the rotor speed on the drone dynamics*/
224  Real fOmegaR = -std::sqrt(std::abs(arrSquaredRotorSpeeds[0])) +
225  std::sqrt(std::abs(arrSquaredRotorSpeeds[1])) +
226  -std::sqrt(std::abs(arrSquaredRotorSpeeds[2])) +
227  std::sqrt(std::abs(arrSquaredRotorSpeeds[3]));
228  /* calculate the drone thrusts and torques */
229  Real fThrust = B *
230  ( arrSquaredRotorSpeeds[0] + arrSquaredRotorSpeeds[1] + arrSquaredRotorSpeeds[2] + arrSquaredRotorSpeeds[3]);
231  Real fTorqueX = (ARM_LENGTH / ROOT_TWO) * B *
232  (-arrSquaredRotorSpeeds[0] - arrSquaredRotorSpeeds[1] + arrSquaredRotorSpeeds[2] + arrSquaredRotorSpeeds[3]);
233  Real fTorqueY = (ARM_LENGTH / ROOT_TWO) * B *
234  (-arrSquaredRotorSpeeds[0] + arrSquaredRotorSpeeds[1] + arrSquaredRotorSpeeds[2] - arrSquaredRotorSpeeds[3]);
235  Real fTorqueZ = D *
236  (-arrSquaredRotorSpeeds[0] + arrSquaredRotorSpeeds[1] - arrSquaredRotorSpeeds[2] + arrSquaredRotorSpeeds[3]);
237  /* saturate the thrust */
238  THRUST_LIMIT.TruncValue(fThrust);
239  /* calculate the system response: acceleration */
240  CVector3 cAcceleration {
241  (std::cos(fYawAngleCorrected) * std::sin(m_cOrientation.GetY()) * std::cos(m_cOrientation.GetX()) +
242  std::sin(fYawAngleCorrected) * std::sin(m_cOrientation.GetX())) * fThrust / MASS,
243  (std::sin(fYawAngleCorrected) * std::sin(m_cOrientation.GetY()) * std::cos(m_cOrientation.GetX()) -
244  std::cos(fYawAngleCorrected) * std::sin(m_cOrientation.GetX())) * fThrust / MASS,
245  -m_cPM3DEngine.GetGravity() + (std::cos(m_cOrientation.GetY()) * std::cos(m_cOrientation.GetX()) * fThrust / MASS)
246  };
247  /* calculate the system response: angular acceleration */
248  CVector3 cAngularAcceleration {
249  (fTorqueX / INERTIA.GetX()) -
250  (JR * m_cAngularVelocity.GetY() * fOmegaR / INERTIA.GetX()) +
251  (m_cAngularVelocity.GetZ() * m_cAngularVelocity.GetY() * (INERTIA.GetY() - INERTIA.GetZ()) / INERTIA.GetX()),
252  (fTorqueY / INERTIA.GetY()) +
253  (JR * m_cAngularVelocity.GetX() * fOmegaR / INERTIA.GetY()) +
254  (m_cAngularVelocity.GetZ() * m_cAngularVelocity.GetX() * (INERTIA.GetZ() - INERTIA.GetX()) / INERTIA.GetY()),
255  (fTorqueZ / INERTIA.GetZ()) +
256  (m_cAngularVelocity.GetY() * m_cAngularVelocity.GetX() * (INERTIA.GetX() - INERTIA.GetY()) / INERTIA.GetZ())
257  };
258  /* gyroscope sensor readings */
259  cAngularAcceleration.Set(
260  cAngularAcceleration.GetX() + m_pcRNG->Gaussian(MEMS_GYRO_NOISE_STD_DEV_X, MEMS_GYRO_NOISE_MEAN) + m_fGyroBias,
261  cAngularAcceleration.GetY() + m_pcRNG->Gaussian(MEMS_GYRO_NOISE_STD_DEV_Y, MEMS_GYRO_NOISE_MEAN) + m_fGyroBias,
262  cAngularAcceleration.GetZ() + m_pcRNG->Gaussian(MEMS_GYRO_NOISE_STD_DEV_Z, MEMS_GYRO_NOISE_MEAN) + m_fGyroBias
263  );
264  m_fGyroBias += m_fAngleRandomWalk * m_pcRNG->Gaussian(MEMS_GYRO_BIAS_STD_DEV, MEMS_GYRO_BIAS_MEAN);
265  /* update the gyro bias angular random walk */
266  m_fAngleRandomWalk *= std::sqrt(GetPM3DEngine().GetPhysicsClockTick());
267  /* update the angular velocity using trapezoid integration */
268  cAngularAcceleration = 0.5 * (m_cAngularAccelerationPrev + cAngularAcceleration);
269  m_cAngularVelocity += cAngularAcceleration * GetPM3DEngine().GetPhysicsClockTick();
270  m_cAngularAccelerationPrev = cAngularAcceleration;
271  /* update the orientation using trapezoid integration */
272  m_cAngularVelocity = 0.5 * (m_cAngularVelocityPrev + m_cAngularVelocity);
273  m_cOrientation += m_cAngularVelocity * GetPM3DEngine().GetPhysicsClockTick();
274  m_cAngularVelocityPrev = m_cAngularVelocity;
275  /* accelerometer sensor readings */
276  cAcceleration.Set(
277  cAcceleration.GetX() + m_pcRNG->Gaussian(MEMS_ACCEL_NOISE_STD_DEV_X, MEMS_ACCEL_NOISE_MEAN) + m_fAccelBias,
278  cAcceleration.GetY() + m_pcRNG->Gaussian(MEMS_ACCEL_NOISE_STD_DEV_Y, MEMS_ACCEL_NOISE_MEAN) + m_fAccelBias,
279  cAcceleration.GetZ() + m_pcRNG->Gaussian(MEMS_ACCEL_NOISE_STD_DEV_Z, MEMS_ACCEL_NOISE_MEAN) + m_fAccelBias
280  );
281  m_fAccelBias += m_fVelocityRandomWalk * m_pcRNG->Gaussian(MEMS_ACCEL_BIAS_STD_DEV, MEMS_ACCEL_BIAS_MEAN);
282  /* update the accel bias velocity random walk */
283  m_fVelocityRandomWalk *= std::sqrt(GetPM3DEngine().GetPhysicsClockTick());
284  /* update the velocity using trapezoid integration */
285  cAcceleration = 0.5 * (m_cAccelerationPrev + cAcceleration);
286  m_cVelocity += cAcceleration * GetPM3DEngine().GetPhysicsClockTick();
287  m_cAccelerationPrev = cAcceleration;
288  /* update the position using trapezoid integration */
289  m_cVelocity = 0.5 * (m_cVelocityPrev + m_cVelocity);
290  m_cPosition += m_cVelocity * GetPM3DEngine().GetPhysicsClockTick();
291  m_cVelocityPrev = m_cVelocity;
292  /* TODO: position and attitude estimation and accel/gyro sensor fusion */
293  }

306  void CPointMass3DDroneModel::MoveTo(const CVector3 &c_position,
307  const CQuaternion &c_orientation) {
308  /* convert to Euler angles */
309  CRadians cYaw, cPitch, cRoll;
310  c_orientation.ToEulerAngles(cYaw, cPitch, cRoll);
311  /* calculate the requested change in position and yaw */
312  Real fDeltaYaw = cYaw.GetValue() - m_cOrientation.GetZ();
313  /* update the home yaw and home position */
314  m_fHomeYawAngle += fDeltaYaw;
315  CVector3 cOffsetPosition(m_cHomePosition - m_cPosition);
316  m_cHomePosition = c_position + cOffsetPosition.RotateZ(CRadians(fDeltaYaw));
317  /* update the position and orientation */
318  m_cPosition = c_position;
319  m_cOrientation.Set(cRoll.GetValue(),
320  cPitch.GetValue(),
321  cYaw.GetValue());
322  /* update the previous orientation target */
323  m_cOrientationTargetPrev.SetZ(m_cOrientationTargetPrev.GetZ() + fDeltaYaw);
324  /* update the space */
325  UpdateEntityStatus();
326  }
