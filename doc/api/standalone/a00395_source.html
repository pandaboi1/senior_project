77  void CBlockEntity::Init(TConfigurationNode& t_tree) {
78  try {
79  /* Initialize the parent */
80  CComposableEntity::Init(t_tree);
81  /* create and initialize the embodied entity */
82  m_pcEmbodiedEntity = new CEmbodiedEntity(this, "body_0");
83  AddComponent(*m_pcEmbodiedEntity);
84  m_pcEmbodiedEntity->Init(GetNode(t_tree, "body"));
85  /* check if this block is movable */
86  bool bMovable = true;
87  GetNodeAttributeOrDefault(t_tree, "movable", bMovable, bMovable);
88  m_pcEmbodiedEntity->SetMovable(bMovable);
89  /* get a reference to the origin anchor */
90  SAnchor& sOriginAnchor = m_pcEmbodiedEntity->GetOriginAnchor();
91  /* create and initialize the tags */
92  m_pcTagEquippedEntity = new CTagEquippedEntity(this, "tags_0");
93  for(const std::tuple<std::string, CVector3, CQuaternion>& c_face : FACE_DESCRIPTORS) {
94  /* add a tag */
95  m_pcTagEquippedEntity->AddTag(std::get<std::string>(c_face),
96  std::get<CVector3>(c_face),
97  std::get<CQuaternion>(c_face),
98  sOriginAnchor,
99  CRadians::PI_OVER_THREE,
100  TAG_SIDE_LENGTH,
101  std::string("0"));
102  }
103  CTagMedium& cTagMedium =
104  CSimulator::GetInstance().GetMedium<CTagMedium>("tags");
105  m_pcTagEquippedEntity->SetMedium(cTagMedium);
106  m_pcTagEquippedEntity->Enable();
107  AddComponent(*m_pcTagEquippedEntity);
108  /* create and initialize the radios */
109  CSimpleRadioMedium& cRadioMedium =
110  CSimulator::GetInstance().GetMedium<CSimpleRadioMedium>("nfc");
111  m_pcSimpleRadioEquippedEntity = new CSimpleRadioEquippedEntity(this, "simple_radios_0");
112  for(const std::tuple<std::string, CVector3, CQuaternion>& c_face : FACE_DESCRIPTORS) {
113  /* add a radio */
114  m_pcSimpleRadioEquippedEntity->AddRadio(std::get<std::string>(c_face),
115  std::get<CVector3>(c_face),
116  sOriginAnchor,
117  cRadioMedium,
118  NFC_TRANSMISSION_RANGE);
119  }
120  m_pcSimpleRadioEquippedEntity->Enable();
121  AddComponent(*m_pcSimpleRadioEquippedEntity);
122  /* create and initialize the directional LEDs */
123  CColor cInitLedColor = CColor::BLACK;
124  GetNodeAttributeOrDefault(t_tree, "init_led_color", cInitLedColor, cInitLedColor);
125  m_pcDirectionalLEDEquippedEntity = new CDirectionalLEDEquippedEntity(this, "directional_leds_0");
126  for(const std::tuple<std::string, CVector3, CQuaternion>& c_face : FACE_DESCRIPTORS) {
127  for(UInt32 un_index = 0; un_index < LED_DESCRIPTORS.size(); un_index++) {
128  /* calculate the LED offset from the origin anchor */
129  CVector3 cPosition(LED_DESCRIPTORS[un_index]);
130  cPosition.Rotate(std::get<CQuaternion>(c_face));
131  cPosition += std::get<CVector3>(c_face);
132  /* create an id for the LED */
133  std::string strId(std::get<std::string>(c_face));
134  strId += ("_" + std::to_string(un_index));
135  /* add the LED */
136  m_pcDirectionalLEDEquippedEntity->AddLED(strId,
137  cPosition,
138  std::get<CQuaternion>(c_face),
139  sOriginAnchor,
140  CRadians::PI_OVER_THREE,
141  cInitLedColor);
142  }
143  }
144  CDirectionalLEDMedium& cDirectionalLEDMedium =
145  CSimulator::GetInstance().GetMedium<CDirectionalLEDMedium>("directional_leds");
146  m_pcDirectionalLEDEquippedEntity->SetMedium(cDirectionalLEDMedium);
147  m_pcDirectionalLEDEquippedEntity->Enable();
148  AddComponent(*m_pcDirectionalLEDEquippedEntity);
149  /* create and initialize the controllable entity */
150  m_pcControllableEntity = new CControllableEntity(this);
151  AddComponent(*m_pcControllableEntity);
152  m_pcControllableEntity->Init(GetNode(t_tree, "controller"));
153  /* update components */
154  UpdateComponents();
155  }
156  catch(CARGoSException& ex) {
157  THROW_ARGOSEXCEPTION_NESTED("Failed to initialize Block", ex);
158  }
159  }
