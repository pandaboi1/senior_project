22  CDynamics3DPiPuckModel::CDynamics3DPiPuckModel(CDynamics3DEngine& c_engine,
23  CPiPuckEntity& c_pipuck) :
24  CDynamics3DMultiBodyObjectModel(c_engine, c_pipuck, 3, false),
25  m_cDifferentialDriveEntity(c_pipuck.GetDifferentialDriveEntity()) {
26  /* get the required collision shapes */
27  std::shared_ptr<btCollisionShape> ptrBodyShape =
28  CDynamics3DShapeManager::RequestCylinder(BODY_HALF_EXTENTS);
29  std::shared_ptr<btCollisionShape> ptrWheelShape =
30  CDynamics3DShapeManager::RequestCylinder(WHEEL_HALF_EXTENTS);
31  /* calculate the inertia of the collision objects */
32  btVector3 cBodyInertia;
33  btVector3 cWheelInertia;
34  ptrBodyShape->calculateLocalInertia(BODY_MASS, cBodyInertia);
35  ptrWheelShape->calculateLocalInertia(WHEEL_MASS, cWheelInertia);
36  /* calculate a btTransform that moves us from the global coordinate system to the
37  local coordinate system */
38  const SAnchor& sOriginAnchor = c_pipuck.GetEmbodiedEntity().GetOriginAnchor();
39  const CQuaternion& cOrientation = sOriginAnchor.Orientation;
40  const CVector3& cPosition = sOriginAnchor.Position;
41  const btTransform& cStartTransform = btTransform(
42  btQuaternion(cOrientation.GetX(),
43  cOrientation.GetZ(),
44  -cOrientation.GetY(),
45  cOrientation.GetW()),
46  btVector3(cPosition.GetX(),
47  cPosition.GetZ(),
48  -cPosition.GetY()));
49  /* create a CAbstractBody::SData structure for each body */
50  CAbstractBody::SData sBodyData(
51  cStartTransform * BODY_OFFSET,
52  BODY_GEOMETRIC_OFFSET,
53  cBodyInertia,
54  BODY_MASS,
55  BODY_FRICTION);
56  CAbstractBody::SData sLeftWheelData(
57  cStartTransform * WHEEL_OFFSET_LEFT,
58  WHEEL_GEOMETRIC_OFFSET,
59  cWheelInertia,
60  WHEEL_MASS,
61  WHEEL_FRICTION);
62  CAbstractBody::SData sRightWheelData(
63  cStartTransform * WHEEL_OFFSET_RIGHT,
64  WHEEL_GEOMETRIC_OFFSET,
65  cWheelInertia,
66  WHEEL_MASS,
67  WHEEL_FRICTION);
68  SAnchor* psBodyAnchor = &c_pipuck.GetEmbodiedEntity().GetAnchor("body");
69  SAnchor* psLeftWheelAnchor = &c_pipuck.GetEmbodiedEntity().GetAnchor("left_wheel");
70  SAnchor* psRightWheelAnchor = &c_pipuck.GetEmbodiedEntity().GetAnchor("right_wheel");
71  /* create the bodies */
72  m_ptrBody = std::make_shared<CBase>(*this, psBodyAnchor, ptrBodyShape, sBodyData);
73  m_ptrLeftWheel = std::make_shared<CLink>(*this, 0, psLeftWheelAnchor, ptrWheelShape, sLeftWheelData);
74  m_ptrRightWheel = std::make_shared<CLink>(*this, 1, psRightWheelAnchor, ptrWheelShape, sRightWheelData);
75  /* copy the bodies to the base class */
76  m_vecBodies = {m_ptrBody, m_ptrLeftWheel, m_ptrRightWheel};
77  /* synchronize with the entity with the space */
78  Reset();
79  }

84  void CDynamics3DPiPuckModel::Reset() {
85  /* reset the base class */
86  CDynamics3DMultiBodyObjectModel::Reset();
87  /* set up wheels */
88  m_cMultiBody.setupRevolute(m_ptrLeftWheel->GetIndex(),
89  m_ptrLeftWheel->GetData().Mass,
90  m_ptrLeftWheel->GetData().Inertia,
91  m_ptrBody->GetIndex(),
92  BODY_TO_WHEEL_LEFT_JOINT_ROTATION,
93  btVector3(0.0, 1.0, 0.0),
94  BODY_TO_WHEEL_LEFT_JOINT_OFFSET,
95  WHEEL_LEFT_TO_BODY_JOINT_OFFSET,
96  true);
97  m_cMultiBody.setupRevolute(m_ptrRightWheel->GetIndex(),
98  m_ptrRightWheel->GetData().Mass,
99  m_ptrRightWheel->GetData().Inertia,
100  m_ptrBody->GetIndex(),
101  BODY_TO_WHEEL_RIGHT_JOINT_ROTATION,
102  btVector3(0.0, 1.0, 0.0),
103  BODY_TO_WHEEL_RIGHT_JOINT_OFFSET,
104  WHEEL_RIGHT_TO_BODY_JOINT_OFFSET,
105  true);
106  /* set up motors for the wheels */
107  m_ptrLeftMotor =
108  std::make_unique<btMultiBodyJointMotor>(&m_cMultiBody,
109  m_ptrLeftWheel->GetIndex(),
110  0.0,
111  WHEEL_MOTOR_MAX_IMPULSE);
112  m_ptrLeftMotor->setRhsClamp(WHEEL_MOTOR_CLAMP);
113  m_ptrRightMotor =
114  std::make_unique<btMultiBodyJointMotor>(&m_cMultiBody,
115  m_ptrRightWheel->GetIndex(),
116  0.0,
117  WHEEL_MOTOR_MAX_IMPULSE);
118  m_ptrRightMotor->setRhsClamp(WHEEL_MOTOR_CLAMP);
119  /* Allocate memory and prepare the btMultiBody */
120  m_cMultiBody.finalizeMultiDof();
121  /* Synchronize with the entity in the space */
122  UpdateEntityStatus();
123  }

130  /* Initialize the bounding box with the base's AABB */
131  m_ptrBody->GetShape().getAabb(m_ptrBody->GetTransform(), cModelAabbMin, cModelAabbMax);
132  /* Write back the bounding box swapping the coordinate systems and the Y component */
133  GetBoundingBox().MinCorner.Set(cModelAabbMin.getX(), -cModelAabbMax.getZ(), cModelAabbMin.getY());
