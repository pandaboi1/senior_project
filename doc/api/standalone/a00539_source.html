39  void CBuilderBotLiftSystemEntity::Update() {
40  /* get the position from the anchor */
41  m_fPosition = m_psAnchor->Position.GetZ() -
42  m_psAnchor->OffsetPosition.GetZ();
43  /* update the limit switches */
44  m_arrLimitSwitches[0] =
45  m_fPosition > UPPER_LIMIT_SWITCH_THRESHOLD ? 1 : 0;
46  m_arrLimitSwitches[1] =
47  m_fPosition < LOWER_LIMIT_SWITCH_THRESHOLD ? 1 : 0;
48  /* step state machine */
49  switch(m_eState) {
50  case EState::INACTIVE:
51  m_fTargetVelocity = 0;
52  break;
53  case EState::POSITION_CONTROL:
54  if(std::fabs(m_fTargetPosition - m_fPosition) < LIFT_ALLOWABLE_ERROR) {
55  m_eState = EState::INACTIVE;
56  m_fTargetVelocity = 0;
57  }
58  else {
59  m_fTargetVelocity = (m_fTargetPosition - m_fPosition) * LIFT_GAIN;
60  if(std::fabs(m_fTargetVelocity) > LIFT_MAX_VELOCITY) {
61  m_fTargetVelocity = std::copysign(LIFT_MAX_VELOCITY, m_fTargetVelocity);
62  }
63  if(std::fabs(m_fTargetVelocity) < LIFT_MIN_VELOCITY) {
64  m_fTargetVelocity = std::copysign(LIFT_MIN_VELOCITY, m_fTargetVelocity);
65  }
66  /* check limit switches */
67  if((m_fTargetVelocity > 0 && m_arrLimitSwitches[0] == 1) ||
68  (m_fTargetVelocity < 0 && m_arrLimitSwitches[1] == 1)) {
69  m_eState = EState::INACTIVE;
70  m_fTargetVelocity = 0;
71  }
72  }
73  break;
74  case EState::CALIBRATION_SEARCH_TOP:
75  m_fTargetVelocity = LIFT_MAX_VELOCITY;
76  if(m_arrLimitSwitches[0] == 1) {
77  m_eState = EState::INACTIVE;
78  }
79  break;
80  case EState::CALIBRATION_SEARCH_BOTTOM:
81  m_fTargetVelocity = -LIFT_MAX_VELOCITY;
82  if(m_arrLimitSwitches[1] == 1) {
83  m_eState = EState::CALIBRATION_SEARCH_TOP;
84  }
85  break;
86  default:
87  THROW_ARGOSEXCEPTION("Unimplemented state reached in the lift system entity");
88  break;
89  }
90  }
