150  void CDroneCamerasSystemDefaultSensor::
151  SSimulatedInterface::Update() {
152  /* increment the interface timestamp */
153  Timestamp += CPhysicsEngine::GetSimulationClockTick();
154  /* clear out the readings from the last update */
155  Tags.clear();
156  /* if the sensor is enabled */
157  if(Enabled) {
158  /* TODO move these sensor parameters to class wide to avoid recalling constructors */
159  CTransformationMatrix3 cWorldToAnchorTransform;
160  CTransformationMatrix3 cWorldToCameraTransform;
161  CVector3 cLookAt, cUp;
162  CVector3 cX, cY, cZ;
163  CVector3 cFarCenter, cNearCenter;
164  std::array<CVector3, 8> arrFrustumExtents;
165  CVector3& cNearTopLeft = arrFrustumExtents[0];
166  CVector3& cNearTopRight = arrFrustumExtents[1];
167  CVector3& cNearBottomLeft = arrFrustumExtents[2];
168  CVector3& cNearBottomRight = arrFrustumExtents[3];
169  CVector3& cFarTopLeft = arrFrustumExtents[4];
170  CVector3& cFarTopRight = arrFrustumExtents[5];
171  CVector3& cFarBottomLeft = arrFrustumExtents[6];
172  CVector3& cFarBottomRight = arrFrustumExtents[7];
173  /* calculate transform matrices */
174  cWorldToAnchorTransform.SetFromComponents(Anchor.Orientation, Anchor.Position);
175  cWorldToCameraTransform = cWorldToAnchorTransform * m_cOffset;
176  m_cCameraToWorldTransform = cWorldToCameraTransform.GetInverse();
177  /* calculate camera direction vectors */
178  m_cCameraPosition = cWorldToCameraTransform.GetTranslationVector();
179  m_cCameraOrientation = Anchor.Orientation * std::get<CQuaternion>(m_tConfiguration);
180  cLookAt = cWorldToCameraTransform * CVector3::Z;
181  cUp = -CVector3::Y;
182  cUp.Rotate(cWorldToCameraTransform.GetRotationMatrix());
183  /* calculate direction vectors */
184  cZ = m_cCameraPosition - cLookAt;
185  cZ.Normalize();
186  cX = cUp;
187  cX.CrossProduct(cZ);
188  cX.Normalize();
189  cY = cZ;
190  cY.CrossProduct(cX);
191  /* calculate frustum coordinates */
192  cNearCenter = m_cCameraPosition - cZ * CAMERA_RANGE_MIN;
193  cFarCenter = m_cCameraPosition - cZ * CAMERA_RANGE_MAX;
194  cNearTopLeft = cNearCenter + (cY * m_fNearPlaneHeight) - (cX * m_fNearPlaneWidth);
195  cNearTopRight = cNearCenter + (cY * m_fNearPlaneHeight) + (cX * m_fNearPlaneWidth);
196  cNearBottomLeft = cNearCenter - (cY * m_fNearPlaneHeight) - (cX * m_fNearPlaneWidth);
197  cNearBottomRight = cNearCenter - (cY * m_fNearPlaneHeight) + (cX * m_fNearPlaneWidth);
198  cFarTopLeft = cFarCenter + (cY * m_fFarPlaneHeight) - (cX * m_fFarPlaneWidth);
199  cFarTopRight = cFarCenter + (cY * m_fFarPlaneHeight) + (cX * m_fFarPlaneWidth);
200  cFarBottomLeft = cFarCenter - (cY * m_fFarPlaneHeight) - (cX * m_fFarPlaneWidth);
201  cFarBottomRight = cFarCenter - (cY * m_fFarPlaneHeight) + (cX * m_fFarPlaneWidth);
202  /* show frustum if enabled by adding outline to the checked rays vector */
203  if(m_cParent.ShowFustrum()) {
204  std::vector<std::pair<bool, CRay3> >& vecCheckedRays =
205  m_cParent.GetControllableEntity().GetCheckedRays();
206  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cNearTopRight));
207  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cNearBottomRight));
208  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cNearBottomLeft));
209  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cNearTopLeft));
210  vecCheckedRays.emplace_back(false, CRay3(cFarTopLeft, cFarTopRight));
211  vecCheckedRays.emplace_back(false, CRay3(cFarTopRight, cFarBottomRight));
212  vecCheckedRays.emplace_back(false, CRay3(cFarBottomRight, cFarBottomLeft));
213  vecCheckedRays.emplace_back(false, CRay3(cFarBottomLeft, cFarTopLeft));
214  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cFarTopLeft));
215  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cFarTopRight));
216  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cFarBottomRight));
217  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cFarBottomLeft));
218  }
219  /* generate a bounding box for the frustum */
220  CVector3 cBoundingBoxMinCorner(cNearCenter);
221  CVector3 cBoundingBoxMaxCorner(cNearCenter);
222  for(const CVector3& c_point : arrFrustumExtents) {
223  if(c_point.GetX() > cBoundingBoxMaxCorner.GetX()) cBoundingBoxMaxCorner.SetX(c_point.GetX());
224  if(c_point.GetX() < cBoundingBoxMinCorner.GetX()) cBoundingBoxMinCorner.SetX(c_point.GetX());
225  if(c_point.GetY() > cBoundingBoxMaxCorner.GetY()) cBoundingBoxMaxCorner.SetY(c_point.GetY());
226  if(c_point.GetY() < cBoundingBoxMinCorner.GetY()) cBoundingBoxMinCorner.SetY(c_point.GetY());
227  if(c_point.GetZ() > cBoundingBoxMaxCorner.GetZ()) cBoundingBoxMaxCorner.SetZ(c_point.GetZ());
228  if(c_point.GetZ() < cBoundingBoxMinCorner.GetZ()) cBoundingBoxMinCorner.SetZ(c_point.GetZ());
229  }
230  cBoundingBoxMaxCorner *= 0.5f;
231  cBoundingBoxMinCorner *= 0.5f;
232  CVector3 cBoundingBoxPosition(cBoundingBoxMaxCorner + cBoundingBoxMinCorner);
233  CVector3 cBoundingBoxHalfExtents(cBoundingBoxMaxCorner - cBoundingBoxMinCorner);
234  /* generate frustum planes */
235  m_arrFrustumPlanes[0].SetFromThreePoints(cNearTopRight, cNearTopLeft, cFarTopLeft);
236  m_arrFrustumPlanes[1].SetFromThreePoints(cNearBottomLeft, cNearBottomRight, cFarBottomRight);
237  m_arrFrustumPlanes[2].SetFromThreePoints(cNearTopLeft, cNearBottomLeft, cFarBottomLeft);
238  m_arrFrustumPlanes[3].SetFromThreePoints(cNearBottomRight, cNearTopRight, cFarBottomRight);
239  m_arrFrustumPlanes[4].SetFromThreePoints(cNearTopLeft, cNearTopRight, cNearBottomRight);
240  m_arrFrustumPlanes[5].SetFromThreePoints(cFarTopRight, cFarTopLeft, cFarBottomLeft);
241  /* set the start of the ray for checking occlusions to the camera's location */
242  m_cOcclusionCheckRay.SetStart(m_cCameraPosition);
243  /* detect tags */
244  m_cParent.GetTagIndex().ForEntitiesInBoxRange(cBoundingBoxPosition,
245  cBoundingBoxHalfExtents,
246  *this);
247  }
248  }

253  bool CDroneCamerasSystemDefaultSensor::
254  SSimulatedInterface::operator()(CTagEntity& c_tag) {
255  if(GetAngleWithCamera(c_tag) > c_tag.GetObservableAngle()) {
256  return true;
257  }
258  std::transform(std::begin(m_arrTagCornerOffsets),
259  std::end(m_arrTagCornerOffsets),
260  std::begin(m_arrTagCorners),
261  [&c_tag] (const CVector3& c_tag_corner_offset) {
262  CVector3 cCorner(c_tag_corner_offset * c_tag.GetSideLength());
263  cCorner.Rotate(c_tag.GetOrientation());
264  return (cCorner + c_tag.GetPosition());
265  });
266  for(const CVector3& c_corner : m_arrTagCorners) {
267  if(IsInsideFrustum(c_corner) == false) {
268  /* no more checks necessary, move on to the next tag */
269  return true;
270  }
271  }
272  for(const CVector3& c_corner : m_arrTagCorners) {
273  m_cOcclusionCheckRay.SetEnd(c_corner);
274  if(GetEmbodiedEntitiesIntersectedByRay(m_vecIntersections, m_cOcclusionCheckRay)) {
275  CEntity& cEntityWithTag = c_tag.GetRootEntity();
276  CEntity& cEntityWithCamera = m_cParent.GetControllableEntity().GetRootEntity();
277  for(const SEmbodiedEntityIntersectionItem& s_item : m_vecIntersections) {
278  CEntity& cIntersectionEntity = s_item.IntersectedEntity->GetRootEntity();
279  if(&cIntersectionEntity != &cEntityWithTag && &cIntersectionEntity != &cEntityWithCamera) {
280  /* corner is occluded */
281  if(m_cParent.ShowTagRays()) {
282  std::vector<std::pair<bool, CRay3> >& vecCheckedRays =
283  m_cParent.GetControllableEntity().GetCheckedRays();
284  vecCheckedRays.emplace_back(true, m_cOcclusionCheckRay);
285  }
286  /* no more checks necessary, move on to the next tag */
287  return true;
288  }
289  }
290  }
291  if(m_cParent.ShowTagRays()) {
292  std::vector<std::pair<bool, CRay3> >& vecCheckedRays =
293  m_cParent.GetControllableEntity().GetCheckedRays();
294  vecCheckedRays.emplace_back(false, m_cOcclusionCheckRay);
295  }
296  }
297  std::transform(std::begin(m_arrTagCorners),
298  std::end(m_arrTagCorners),
299  std::begin(m_arrTagCornerPixels),
300  [this] (const CVector3& c_tag_corner) {
301  return ProjectOntoSensor(c_tag_corner);
302  });
303  const CVector2& cCenterPixel = ProjectOntoSensor(c_tag.GetPosition());
304  /* try to convert tag payload to an unsigned integer */
305  UInt32 unId = 0;
306  try {
307  std::string strId(c_tag.GetPayload());
308  auto itRemove =
309  std::remove_if(std::begin(strId),
310  std::end(strId),
311  [] (char ch) {
312  return (std::isdigit(ch) == 0);
313  });
314  strId.erase(itRemove, std::end(strId));
315  unId = std::stoul(strId);
316  }
317  catch(const std::logic_error& err_logic) {}
318  /* TODO: Update the position and orientation calculations to avoid the use of matrices */
319  CVector3 cTagPosition = m_cCameraToWorldTransform * c_tag.GetPosition();
320  /* Direction of the tag should be pointing inside of the tag */
321  CQuaternion cTagOrientation = m_cCameraOrientation.Inverse() * c_tag.GetOrientation() * CQuaternion(CRadians::PI, CVector3::X);
322  /* transfer readings to the control interface */
323  Tags.emplace_back(unId, cTagPosition, cTagOrientation, cCenterPixel, m_arrTagCornerPixels);
324  return true;
325  }
