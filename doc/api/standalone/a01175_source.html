55  void CSimpleRadiosDefaultActuator::Update() {
56  for(size_t i = 0; i < m_vecInterfaces.size(); ++i) {
57  CSimpleRadioEntity& cRadio = m_pcSimpleRadioEquippedEntity->GetRadio(i);
58  /* Create operation instance */
59  CSendOperation cSendOperation(cRadio, m_vecInterfaces[i].Messages);
60  /* Calculate the coarse range of the transmitting radio */
61  CVector3 cRange(1.0f, 1.0f, 1.0f);
62  cRange *= cRadio.GetRange();
63  /* Get positional index */
64  CPositionalIndex<CSimpleRadioEntity>* pcRadioIndex =
65  &(cRadio.GetMedium().GetIndex());
66  /* Transmit the messages to receiving radios in the space */
67  pcRadioIndex->ForEntitiesInBoxRange(cRadio.GetPosition(), cRange, cSendOperation);
68  /* Flush messages from the control interface */
69  m_vecInterfaces[i].Messages.clear();
70  }
