106  void CQTOpenGLBlock::Draw(CBlockEntity& c_entity) {
107  /* buffers for converting angles */
108  CRadians cZAngle, cYAngle, cXAngle;
109  /* update the LED materials */
110  const CDirectionalLEDEquippedEntity& c_leds = c_entity.GetDirectionalLEDEquippedEntity();
111  for(UInt32 un_material_idx = 0;
112  un_material_idx < m_arrLEDs.size();
113  un_material_idx++) {
114  const CColor& cColor = c_leds.GetLED(un_material_idx).GetColor();
115  std::array<GLfloat, 4> arrColor = {
116  cColor.GetRed() / 255.0f,
117  cColor.GetGreen() / 255.0f,
118  cColor.GetBlue() / 255.0f,
119  1.0f
120  };
121  m_arrLEDs[un_material_idx]->Emission = arrColor;
122  }
123  /* move to the coordinate system of the block */
124  const CVector3& cPosition = c_entity.GetEmbodiedEntity().GetOriginAnchor().Position;
125  const CQuaternion& cOrientation = c_entity.GetEmbodiedEntity().GetOriginAnchor().Orientation;
126  cOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
127  glPushMatrix();
128  glTranslatef(cPosition.GetX(), cPosition.GetY(), cPosition.GetZ());
129  glRotatef(ToDegrees(cXAngle).GetValue(), 1.0f, 0.0f, 0.0f);
130  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
131  glRotatef(ToDegrees(cZAngle).GetValue(), 0.0f, 0.0f, 1.0f);
132  /* draw the body */
133  m_cBlockModel.Draw();
134  /* draw tags */
135  for(const CTagEquippedEntity::SInstance& s_instance :
136  c_entity.GetTagEquippedEntity().GetInstances()) {
137  /* the texture of the tag contains the white border which isn't actually part
138  of the tag. Stretching the texture by 25% fixes this problem */
139  Real fScaling = s_instance.Tag.GetSideLength() * 1.25f;
140  const CVector3& cTagPosition = s_instance.PositionOffset;
141  const CQuaternion& cTagOrientation = s_instance.OrientationOffset;
142  cTagOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
143  glPushMatrix();
144  glTranslatef(cTagPosition.GetX(),
145  cTagPosition.GetY(),
146  cTagPosition.GetZ());
147  glRotatef(ToDegrees(cXAngle).GetValue(), 1.0f, 0.0f, 0.0f);
148  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
149  glRotatef(ToDegrees(cZAngle).GetValue(), 0.0f, 0.0f, 1.0f);
150  glScalef(fScaling, fScaling, 1.0f);
151  glCallList(m_unTagList);
152  glPopMatrix();
153  }
154  glPopMatrix();
155  }
