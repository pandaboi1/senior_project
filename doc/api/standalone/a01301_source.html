49  void CPiPuckRangefindersDefaultSensor::Update() {
50  /* buffers */
51  CRay3 cSensorRay;
52  CVector3 cRayStart, cRayEnd;
53  SEmbodiedEntityIntersectionItem sIntersection;
54  /* go through the sensors */
55  for(SSimulatedInterface& s_interface : m_vecSimulatedInterfaces) {
56  cRayStart = std::get<CVector3>(s_interface.Configuration);
57  cRayStart.Rotate(s_interface.Anchor.Orientation);
58  cRayStart += s_interface.Anchor.Position;
59  cRayEnd = CVector3::Z * std::get<Real>(s_interface.Configuration);
60  cRayEnd.Rotate(std::get<CQuaternion>(s_interface.Configuration));
61  cRayEnd.Rotate(s_interface.Anchor.Orientation);
62  cRayEnd += cRayStart;
63  cSensorRay.Set(cRayStart,cRayEnd);
64  /* Get the closest intersection */
65  if(GetClosestEmbodiedEntityIntersectedByRay(sIntersection, cSensorRay)) {
66  /* There is an intersection */
67  if(m_bShowRays) {
68  m_pcControllableEntity->AddIntersectionPoint(cSensorRay, sIntersection.TOnRay);
69  m_pcControllableEntity->AddCheckedRay(true, cSensorRay);
70  }
71  s_interface.Proximity =
72  ConvertToMeters(cSensorRay.GetDistance(sIntersection.TOnRay));
73  }
74  else {
75  /* No intersection */
76  s_interface.Proximity = std::get<Real>(s_interface.Configuration);
77  if(m_bShowRays) {
78  m_pcControllableEntity->AddCheckedRay(false, cSensorRay);
79  }
80  }
81  /* not implemented for the moment */
82  s_interface.Illuminance = 0.0;
83  }
84  }
