44  /* Update LED materials */
45  const CDirectionalLEDEquippedEntity& c_leds = c_entity.GetDirectionalLEDEquippedEntity();

51  cColor.GetRed() / 255.0f,
52  cColor.GetGreen() / 255.0f,
53  cColor.GetBlue() / 255.0f,
54  1.0f
55  };
56  m_arrLEDs[un_material_idx]->Emission = arrColor;
57  }
58  /* buffers for converting from quaternions to euler angles */

61  const CVector3& cPosition = c_entity.GetEmbodiedEntity().GetOriginAnchor().Position;

63  cOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
64  glPushMatrix();

66  glRotatef(ToDegrees(cXAngle).GetValue(), 1.0f, 0.0f, 0.0f);
67  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
68  glRotatef(ToDegrees(cZAngle).GetValue(), 0.0f, 0.0f, 1.0f);
69  m_cDroneModel.Draw();

71  UInt32 unPropellerAngle =
