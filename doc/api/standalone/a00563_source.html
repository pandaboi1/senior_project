96  void CQTOpenGLBuilderBot::Draw(const CBuilderBotEntity& c_entity) {
97  /* update LED materials */
98  const CDirectionalLEDEquippedEntity& c_leds = c_entity.GetDirectionalLEDEquippedEntity();
99  for(UInt32 un_material_idx = 0;
100  un_material_idx < m_arrLEDs.size();
101  un_material_idx++) {
102  const CColor& cColor = c_leds.GetLED(un_material_idx).GetColor();
103  std::array<GLfloat, 4> arrColor = {
104  cColor.GetRed() / 255.0f,
105  cColor.GetGreen() / 255.0f,
106  cColor.GetBlue() / 255.0f,
107  1.0f
108  };
109  m_arrLEDs[un_material_idx]->Emission = arrColor;
110  }
111  /* buffers for converting to euler angles */
112  CRadians cZAngle, cYAngle, cXAngle;
113  /* draw the robot */
114  const CVector3& cPosition = c_entity.GetEmbodiedEntity().GetOriginAnchor().Position;
115  const CQuaternion& cOrientation = c_entity.GetEmbodiedEntity().GetOriginAnchor().Orientation;
116  cOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
117  glPushMatrix();
118  glTranslatef(cPosition.GetX(), cPosition.GetY(), cPosition.GetZ());
119  glRotatef(ToDegrees(cXAngle).GetValue(), 1.0f, 0.0f, 0.0f);
120  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
121  glRotatef(ToDegrees(cZAngle).GetValue(), 0.0f, 0.0f, 1.0f);
122  m_cBuilderBotModel.Draw();
123  /* draw tags */
124  for(const CTagEquippedEntity::SInstance& s_instance :
125  c_entity.GetTagEquippedEntity().GetInstances()) {
126  /* the texture of the tag contains the white border which isn't actually part
127  of the tag. Stretching the texture by 25% fixes this problem */
128  Real fScaling = s_instance.Tag.GetSideLength() * 1.25f;
129  const CVector3& cTagPosition = s_instance.PositionOffset;
130  const CQuaternion& cTagOrientation = s_instance.OrientationOffset;
131  cTagOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
132  glPushMatrix();
133  glTranslatef(cTagPosition.GetX(),
134  cTagPosition.GetY(),
135  cTagPosition.GetZ());
136  glRotatef(ToDegrees(cXAngle).GetValue(), 1.0f, 0.0f, 0.0f);
137  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
138  glRotatef(ToDegrees(cZAngle).GetValue(), 0.0f, 0.0f, 1.0f);
139  glScalef(fScaling, fScaling, 1.0f);
140  glCallList(m_unTagList);
141  glPopMatrix();
142  }
143  glPopMatrix();
144  /* draw the manipulator */
145  const CVector3& cManipulatorPosition = c_entity.GetLiftSystemEntity().GetAnchor().Position;
146  const CQuaternion& cManipulatorOrientation = c_entity.GetLiftSystemEntity().GetAnchor().Orientation;
147  cManipulatorOrientation.ToEulerAngles(cZAngle, cYAngle, cXAngle);
148  glPushMatrix();

151  glRotatef(ToDegrees(cYAngle).GetValue(), 0.0f, 1.0f, 0.0f);
