102  void CBuilderBotRangefindersDefaultSensor::Update() {
103  class CRangefinderOperation : public CPositionalIndex<CLEDEntity>::COperation {
104  public:
105  CRangefinderOperation(SSimulatedInterface& s_interface) :
106  m_sInterface(s_interface),
107  m_fTotalLight(0.0) {
108  m_cSensorPosition = m_sInterface.PositionOffset;
109  m_cSensorPosition.Rotate(m_sInterface.Anchor.Orientation);
110  m_cSensorPosition += m_sInterface.Anchor.Position;
111  m_cSensorDirection = CVector3::Z * m_sInterface.Range;
112  m_cSensorDirection.Rotate(m_sInterface.OrientationOffset);
113  m_cSensorDirection.Rotate(m_sInterface.Anchor.Orientation);
114  m_cSensorRay.Set(m_cSensorPosition, m_cSensorPosition + m_cSensorDirection);
115  /* Compute reading */
116  /* Get the closest intersection */
117  if(GetClosestEmbodiedEntityIntersectedByRay(m_sIntersection, m_cSensorRay)) {
118  /* There is an intersection */
119  if(m_sInterface.ShowRays) {
120  m_sInterface.ControllableEntity.AddIntersectionPoint(m_cSensorRay, m_sIntersection.TOnRay);
121  m_sInterface.ControllableEntity.AddCheckedRay(true, m_cSensorRay);
122  }
123  Real fDistance = m_cSensorRay.GetDistance(m_sIntersection.TOnRay);
124  m_sInterface.Proximity =
125  CBuilderBotRangefindersDefaultSensor::ConvertToMeters(fDistance);
126  }
127  else {
128  /* No intersection */
129  m_sInterface.Proximity = 0.05;
130  if(m_sInterface.ShowRays) {
131  m_sInterface.ControllableEntity.AddCheckedRay(false, m_cSensorRay);
132  }
133  }
134  }
135  /* destructor writes back the total illuminance */
136  ~CRangefinderOperation() {
137  m_fTotalLight = std::min(m_fTotalLight, static_cast<Real>(UINT8_MAX));
138  m_sInterface.Illuminance = m_fTotalLight / static_cast<Real>(UINT8_MAX);
139  }
140  /* operator() accumulates the light measurement from different sources */
141  virtual bool operator()(CLEDEntity& c_light) {
142  Real fBrightness = c_light.GetColor().ToGrayScale();
143  /* decrease fBrightness with respect to inverse square law */
144  fBrightness *= 1.0 / (1.0 + SquareDistance(c_light.GetPosition(), m_cSensorPosition));
145  /* decrease fBrightness with respect to the angle between the light source and the sensor */
146  CVector3 cSensorToLight(c_light.GetPosition() - m_cSensorPosition);
147  Real fCosine =
148  (m_cSensorDirection.DotProduct(cSensorToLight) /
149  (m_cSensorDirection.Length() * cSensorToLight.Length()));
150  fBrightness *= (fCosine + 1.0) / 2.0;
151  /* the following lines were written to emulate the reduction in light due to the shadows
152  cast by emodied entities in the simulation, uncomment at your own risk */
153  //m_cSensorRay.SetEnd(c_light.GetPosition());
154  //if(GetClosestEmbodiedEntityIntersectedByRay(m_sIntersection, m_cSensorRay, m_sInterface.EmbodiedEntity)) {
155  // fBrightness *= Abs(0.5 - m_sIntersection.TOnRay);
156  //}
157  m_fTotalLight += fBrightness;
158  return true;
159  }
160  private:
161  /* local data */
162  SSimulatedInterface& m_sInterface;
163  Real m_fTotalLight;
164  CRay3 m_cSensorRay;
165  CVector3 m_cSensorPosition, m_cSensorDirection;
166  SEmbodiedEntityIntersectionItem m_sIntersection;
167  };
168  /* for each sensor */
169  for(SSimulatedInterface& s_simulated_interface : m_vecSimulatedInterfaces) {
170  /* COperation::COperation performs the proximity measurement */
171  CRangefinderOperation cRangefinderOperation(s_simulated_interface);
172  /* COperation::operator() performs the light measurement */
173  if(m_pcLightIndex != nullptr) {
174  m_pcLightIndex->ForAllEntities(cRangefinderOperation);
175  }
176  }
177  }
