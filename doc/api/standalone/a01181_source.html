64  void CSimpleRadiosDefaultSensor::Update() {
65  /* sensor is disabled--nothing to do */
66  if (IsDisabled()) {
67  return;
68  }
69  for(size_t i = 0; i < m_pcSimpleRadioEquippedEntity->GetInstances().size(); ++i) {
70  CSimpleRadioEntity& cRadio = m_pcSimpleRadioEquippedEntity->GetRadio(i);
71  /* Clear messages in the interface */
72  m_vecInterfaces[i].Messages.clear();
73  /* Copy messages from the radio entity to the control interface */
74  for(const std::pair<CVector3, CByteArray>& c_message : cRadio.GetMessages()) {
75  m_vecInterfaces[i].Messages.emplace_back(c_message.second);
76  if(m_bShowRays) {
77  CRay3 cRay(c_message.first, cRadio.GetPosition());
78  m_pcControllableEntity->GetCheckedRays().emplace_back(!c_message.second.Empty(), cRay);
79  }
80  }
81  /* Clear messages in the radio entity */
82  cRadio.GetMessages().clear();
83  }
84  }
