86  void CDynamics3DSRoCSPlugin::Update() {
87  /* calculate the magnet offsets and positions for each block */
88  for(SBlock& s_block : m_vecBlocks) {
89  s_block.InteractingWithEndEffector = false;
90  /* for each of the eight magnets, calculate rotated offsets and positions */
91  for(UInt32 un_index = 0; un_index < 8; un_index++) {
92  s_block.Magnets[un_index].RotatedOffset =
93  quatRotate(s_block.Body->GetTransform().getRotation(), m_arrMagnetOffsets[un_index]);
94  s_block.Magnets[un_index].Position =
95  s_block.Magnets[un_index].RotatedOffset + s_block.Body->GetTransform().getOrigin();
96  }
97  }
98  /* calculate the electromagnet offsets for each end effector */
99  for(SEndEffector& s_end_effector : m_vecEndEffectors) {
100  /* for each of the four electromagnets */
101  for(UInt32 un_index = 0; un_index < 4; un_index++) {
102  s_end_effector.Electromagnets[un_index].Position =
103  s_end_effector.Body->GetTransform().getOrigin();
104  s_end_effector.Electromagnets[un_index].Position +=
105  quatRotate(s_end_effector.Body->GetTransform().getRotation(), m_arrElectromagnetOffsets[un_index]);
106  }
107  }
108  /* force between end effectors and blocks */
109  for(SBlock& s_block : m_vecBlocks) {
110  for(SEndEffector& s_end_effector : m_vecEndEffectors) {
111  /* get the field */
112  Real fElectromagnetField = s_end_effector.Body->GetData().Dipoles[0].GetField().getY();
113  /* calculate the force */
114  btScalar fForce = ELECTROMAGNET_FORCE_COEFFICIENT * s_block.Body->GetData().Mass;
115  /* calculate the range */
116  Real fRange = fElectromagnetField * ELECTROMAGNET_RANGE_COEFFICIENT;
117  if(fRange > ELECTROMAGNET_RANGE_MAX)
118  fRange = ELECTROMAGNET_RANGE_MAX;
119  /* apply force only if there are four pairs of magnets in range of each other */
120  UInt32 unInRangeCount = 0;
121  for(const SBlock::SMagnet& s_magnet : s_block.Magnets) {
122  for(const SEndEffector::SElectromagnet& s_electromagnet : s_end_effector.Electromagnets) {
123  if(s_magnet.Position.distance(s_electromagnet.Position) < fRange) {
124  unInRangeCount++;
125  }
126  }
127  }
128  if(unInRangeCount < 4) {
129  /* not enough points in range */
130  continue;
131  }
132  else {
133  /* apply force on the block */
134  for(const SBlock::SMagnet& s_magnet : s_block.Magnets) {
135  for(const SEndEffector::SElectromagnet& s_electromagnet : s_end_effector.Electromagnets) {
136  btScalar fDistance = s_magnet.Position.distance(s_electromagnet.Position);
137  btVector3 cDirectionVector = (s_electromagnet.Position - s_magnet.Position).normalize();
138  if(fDistance < fRange) {
139  /* force and torque for box */
140  s_block.Body->ApplyForce(cDirectionVector * fForce,
141  s_magnet.RotatedOffset);
142  s_block.InteractingWithEndEffector = true;
143  }
144  }
145  }
146  }
147  }
148  }
149  /* force between blocks */
150  if(m_vecBlocks.size() < 2) {
151  /* Nothing to do for blocks */
152  return;
153  }
154  /* for each pair of blocks */
155  for(std::vector<SBlock>::iterator itBlock0 = std::begin(m_vecBlocks);
156  itBlock0 != (std::end(m_vecBlocks) - 1);
157  ++itBlock0) {
158  if(m_bDisableBlockMagnetsDuringManipulation &&
159  itBlock0->InteractingWithEndEffector) {
160  continue;
161  }
162  for(std::vector<SBlock>::iterator itBlock1 = std::next(itBlock0, 1);
163  itBlock1 != std::end(m_vecBlocks);
164  ++itBlock1) {
165  if(m_bDisableBlockMagnetsDuringManipulation &&
166  itBlock1->InteractingWithEndEffector) {
167  continue;
168  }
169  /* if a box is not movable, its mass is zero by default,
170  * so if one is zero, use the mass of the other one */
171  btScalar fForce = MAGNET_FORCE_COEFFICIENT * itBlock0->Body->GetData().Mass;
172  if(fForce == 0)
173  fForce = MAGNET_FORCE_COEFFICIENT * itBlock1->Body->GetData().Mass;
174  /* apply force only if there are four pairs in range */
175  UInt32 unInRangeCount = 0;
176  for(const SBlock::SMagnet& s_magnet : itBlock0->Magnets) {
177  const btVector3& cMagnetPosition0 = s_magnet.Position;
178  for(const SBlock::SMagnet& s_magnet : itBlock1->Magnets) {
179  const btVector3& cMagnetPosition1 = s_magnet.Position;
180  if(cMagnetPosition0.distance(cMagnetPosition1) < MAGNET_DISTANCE_MAX) {
181  unInRangeCount++;
182  }
183  }
184  }
185  if(unInRangeCount < 4) {
186  continue;
187  }
188  /* apply force on both blocks */
189  for(const SBlock::SMagnet& s_magnet_0 : itBlock0->Magnets) {
190  const btVector3& cMagnetPosition0 = s_magnet_0.Position;
191  for(const SBlock::SMagnet& s_magnet_1 : itBlock1->Magnets) {
192  const btVector3& cMagnetPosition1 = s_magnet_1.Position;
193  /* distance and relative vector */
194  btScalar fDistance = cMagnetPosition0.distance(cMagnetPosition1);
195  btVector3 cDirectionVector01 = (cMagnetPosition1 - cMagnetPosition0).normalize();
196  btVector3 cDirectionVector10 = -cDirectionVector01;
197  /* apply force with respect to distance */
198  if(fDistance < MAGNET_DISTANCE_MAX) {
199  if(fDistance > MAGNET_DISTANCE_CONTACT) {
200  fForce *= (MAGNET_DISTANCE_MAX - fDistance) /
201  (MAGNET_DISTANCE_MAX - MAGNET_DISTANCE_CONTACT);
202  }
203  /* force and torque for magnet 0 */
204  itBlock0->Body->ApplyForce(cDirectionVector01 * fForce,
205  s_magnet_0.RotatedOffset);
206  /* force and torque for magnet 1 */
207  itBlock1->Body->ApplyForce(cDirectionVector10 * fForce,
208  s_magnet_1.RotatedOffset);
209  }
210  }
211  }
212  }
213  }
214  }
