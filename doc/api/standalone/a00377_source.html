602  void CLuaUtility::LuaSerializeTable(CByteArray& c_data,
603  lua_State* pt_lua_state,
604  SInt32 n_table_offset) {
605  lua_pushnil(pt_lua_state);
606  /* if we are indexing from the top of the stack (negative numbers) then we need to decrement
607  n_table_offset to compensate for the nil that we just pushed onto the stack */
608  if(n_table_offset < 0) {
609  n_table_offset -= 1;
610  }
611  while(lua_next(pt_lua_state, n_table_offset)) {
612  /* write the key */
613  SInt8 nKeyTypeId = static_cast<UInt8>(lua_type(pt_lua_state, -2));
614  SInt8 nValueTypeId = static_cast<UInt8>(lua_type(pt_lua_state, -1));
615  CByteArray cKeyValueBytes;
616  cKeyValueBytes << nKeyTypeId;
617  switch(nKeyTypeId) {
618  case LUA_TSTRING:
619  cKeyValueBytes << lua_tostring(pt_lua_state, -2);
620  break;
621  case LUA_TNUMBER:
622  cKeyValueBytes << lua_tonumber(pt_lua_state, -2);
