273  void CDynamics3DBuilderBotModel::CalculateBoundingBox() {
274  btVector3 cModelAabbMin, cModelAabbMax, cLinkAabbMin, cLinkAabbMax;
275  /* Initialize the bounding box with the base's AABB */
276  m_ptrLowerBase->GetShape().getAabb(m_ptrLowerBase->GetTransform(), cModelAabbMin, cModelAabbMax);
277  /* Extend AABB to include other bodies */
278  for(const std::shared_ptr<CLink>& ptr_link : {std::cref(m_ptrLiftColumn), std::cref(m_ptrEndEffector)}) {
279  /* Get the axis aligned bounding box for the current body */
280  ptr_link->GetShape().getAabb(ptr_link->GetTransform(), cLinkAabbMin, cLinkAabbMax);
281  /* Update minimum corner */
282  cModelAabbMin.setX(cModelAabbMin.getX() < cLinkAabbMin.getX() ? cModelAabbMin.getX() : cLinkAabbMin.getX());
283  cModelAabbMin.setY(cModelAabbMin.getY() < cLinkAabbMin.getY() ? cModelAabbMin.getY() : cLinkAabbMin.getY());
284  cModelAabbMin.setZ(cModelAabbMin.getZ() < cLinkAabbMin.getZ() ? cModelAabbMin.getZ() : cLinkAabbMin.getZ());
285  /* Update maximum corner */
286  cModelAabbMax.setX(cModelAabbMax.getX() > cLinkAabbMax.getX() ? cModelAabbMax.getX() : cLinkAabbMax.getX());
287  cModelAabbMax.setY(cModelAabbMax.getY() > cLinkAabbMax.getY() ? cModelAabbMax.getY() : cLinkAabbMax.getY());
288  cModelAabbMax.setZ(cModelAabbMax.getZ() > cLinkAabbMax.getZ() ? cModelAabbMax.getZ() : cLinkAabbMax.getZ());
289  }
290  /* Write back the bounding box swapping the coordinate systems and the Y component */
291  GetBoundingBox().MinCorner.Set(cModelAabbMin.getX(), -cModelAabbMax.getZ(), cModelAabbMin.getY());
292  GetBoundingBox().MaxCorner.Set(cModelAabbMax.getX(), -cModelAabbMin.getZ(), cModelAabbMax.getY());
293  }
