104  void CPiPuckFrontCameraDefaultSensor::Update() {
105  /* clear out the readings from the last update */
106  m_vecTags.clear();
107  m_vecLedCache.clear();
108  /* increment the timestamp */
109  m_fTimestamp += CPhysicsEngine::GetSimulationClockTick();
110  /* if the sensor is enabled */
111  if(IsEnabled()) {
112  /* TODO move these sensor parameters to class wide to avoid recalling constructors */
113  CTransformationMatrix3 cWorldToAnchorTransform;
114  CTransformationMatrix3 cWorldToCameraTransform;
115  CVector3 cLookAt, cUp;
116  CVector3 cX, cY, cZ;
117  CVector3 cFarCenter, cNearCenter;
118  std::array<CVector3, 8> arrFrustumExtents;
119  CVector3& cNearTopLeft = arrFrustumExtents[0];
120  CVector3& cNearTopRight = arrFrustumExtents[1];
121  CVector3& cNearBottomLeft = arrFrustumExtents[2];
122  CVector3& cNearBottomRight = arrFrustumExtents[3];
123  CVector3& cFarTopLeft = arrFrustumExtents[4];
124  CVector3& cFarTopRight = arrFrustumExtents[5];
125  CVector3& cFarBottomLeft = arrFrustumExtents[6];
126  CVector3& cFarBottomRight = arrFrustumExtents[7];
127  /* calculate transform matrices */
128  SAnchor& s_anchor = m_pcEmbodiedEntity->GetOriginAnchor();
129  cWorldToAnchorTransform.SetFromComponents(s_anchor.Orientation, s_anchor.Position);
130  cWorldToCameraTransform = cWorldToAnchorTransform * m_cOffset;
131  m_cCameraToWorldTransform = cWorldToCameraTransform.GetInverse();
132  /* calculate camera direction vectors */
133  m_cCameraPosition = cWorldToCameraTransform.GetTranslationVector();
134  m_cCameraOrientation = s_anchor.Orientation *
135  CCI_PiPuckFrontCameraSensor::m_cOrientationOffset;
136  cLookAt = cWorldToCameraTransform * CVector3::Z;
137  cUp = -CVector3::Y;
138  cUp.Rotate(cWorldToCameraTransform.GetRotationMatrix());
139  /* calculate direction vectors */
140  cZ = m_cCameraPosition - cLookAt;
141  cZ.Normalize();
142  cX = cUp;
143  cX.CrossProduct(cZ);
144  cX.Normalize();
145  cY = cZ;
146  cY.CrossProduct(cX);
147  /* calculate frustum coordinates */
148  cNearCenter = m_cCameraPosition - cZ * CAMERA_RANGE_MIN;
149  cFarCenter = m_cCameraPosition - cZ * CAMERA_RANGE_MAX;
150  cNearTopLeft = cNearCenter + (cY * m_fNearPlaneHeight) - (cX * m_fNearPlaneWidth);
151  cNearTopRight = cNearCenter + (cY * m_fNearPlaneHeight) + (cX * m_fNearPlaneWidth);
152  cNearBottomLeft = cNearCenter - (cY * m_fNearPlaneHeight) - (cX * m_fNearPlaneWidth);
153  cNearBottomRight = cNearCenter - (cY * m_fNearPlaneHeight) + (cX * m_fNearPlaneWidth);
154  cFarTopLeft = cFarCenter + (cY * m_fFarPlaneHeight) - (cX * m_fFarPlaneWidth);
155  cFarTopRight = cFarCenter + (cY * m_fFarPlaneHeight) + (cX * m_fFarPlaneWidth);
156  cFarBottomLeft = cFarCenter - (cY * m_fFarPlaneHeight) - (cX * m_fFarPlaneWidth);
157  cFarBottomRight = cFarCenter - (cY * m_fFarPlaneHeight) + (cX * m_fFarPlaneWidth);
158  /* show frustum if enabled by adding outline to the checked rays vector */
159  if(m_bShowFrustum) {
160  std::vector<std::pair<bool, CRay3> >& vecCheckedRays = m_pcControllableEntity->GetCheckedRays();
161  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cNearTopRight));
162  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cNearBottomRight));
163  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cNearBottomLeft));
164  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cNearTopLeft));
165  vecCheckedRays.emplace_back(false, CRay3(cFarTopLeft, cFarTopRight));
166  vecCheckedRays.emplace_back(false, CRay3(cFarTopRight, cFarBottomRight));
167  vecCheckedRays.emplace_back(false, CRay3(cFarBottomRight, cFarBottomLeft));
168  vecCheckedRays.emplace_back(false, CRay3(cFarBottomLeft, cFarTopLeft));
169  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cFarTopLeft));
170  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cFarTopRight));
171  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cFarBottomRight));
172  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cFarBottomLeft));
173  }
174  /* generate a bounding box for the frustum */
175  CVector3 cBoundingBoxMinCorner(cNearCenter);
176  CVector3 cBoundingBoxMaxCorner(cNearCenter);
177  for(const CVector3& c_point : arrFrustumExtents) {
178  if(c_point.GetX() > cBoundingBoxMaxCorner.GetX()) cBoundingBoxMaxCorner.SetX(c_point.GetX());
179  if(c_point.GetX() < cBoundingBoxMinCorner.GetX()) cBoundingBoxMinCorner.SetX(c_point.GetX());
180  if(c_point.GetY() > cBoundingBoxMaxCorner.GetY()) cBoundingBoxMaxCorner.SetY(c_point.GetY());
181  if(c_point.GetY() < cBoundingBoxMinCorner.GetY()) cBoundingBoxMinCorner.SetY(c_point.GetY());
182  if(c_point.GetZ() > cBoundingBoxMaxCorner.GetZ()) cBoundingBoxMaxCorner.SetZ(c_point.GetZ());
183  if(c_point.GetZ() < cBoundingBoxMinCorner.GetZ()) cBoundingBoxMinCorner.SetZ(c_point.GetZ());
184  }
185  cBoundingBoxMaxCorner *= 0.5f;
186  cBoundingBoxMinCorner *= 0.5f;
187  CVector3 cBoundingBoxPosition(cBoundingBoxMaxCorner + cBoundingBoxMinCorner);
188  CVector3 cBoundingBoxHalfExtents(cBoundingBoxMaxCorner - cBoundingBoxMinCorner);
189  /* generate frustum planes */
190  m_arrFrustumPlanes[0].SetFromThreePoints(cNearTopRight, cNearTopLeft, cFarTopLeft);
191  m_arrFrustumPlanes[1].SetFromThreePoints(cNearBottomLeft, cNearBottomRight, cFarBottomRight);
192  m_arrFrustumPlanes[2].SetFromThreePoints(cNearTopLeft, cNearBottomLeft, cFarBottomLeft);
193  m_arrFrustumPlanes[3].SetFromThreePoints(cNearBottomRight, cNearTopRight, cFarBottomRight);
194  m_arrFrustumPlanes[4].SetFromThreePoints(cNearTopLeft, cNearTopRight, cNearBottomRight);
195  m_arrFrustumPlanes[5].SetFromThreePoints(cFarTopRight, cFarTopLeft, cFarBottomLeft);
196  /* set the start of the ray for checking occlusions to the camera's location */
197  m_cOcclusionCheckRay.SetStart(m_cCameraPosition);
198  /* detect tags */
199  if(m_pcTagIndex) {
200  m_pcTagIndex->ForEntitiesInBoxRange(cBoundingBoxPosition,
201  cBoundingBoxHalfExtents,
202  *this);
203  }
204  /* detect directional LEDs */

207  cBoundingBoxHalfExtents,

228  bool CPiPuckFrontCameraDefaultSensor::operator()(CTagEntity& c_tag) {
229  if(GetAngleWithCamera(c_tag) > c_tag.GetObservableAngle()) {
230  return true;
231  }
232  std::transform(std::begin(m_arrTagCornerOffsets),
233  std::end(m_arrTagCornerOffsets),
234  std::begin(m_arrTagCorners),
235  [&c_tag] (const CVector3& c_tag_corner_offset) {
236  CVector3 cCorner(c_tag_corner_offset * c_tag.GetSideLength());
237  cCorner.Rotate(c_tag.GetOrientation());
238  return (cCorner + c_tag.GetPosition());
239  });
240  for(const CVector3& c_corner : m_arrTagCorners) {
241  if(IsInsideFrustum(c_corner) == false) {
242  /* no more checks necessary, move on to the next tag */
243  return true;
244  }
245  }
246  for(const CVector3& c_corner : m_arrTagCorners) {
247  m_cOcclusionCheckRay.SetEnd(c_corner);
248  if(GetClosestEmbodiedEntityIntersectedByRay(m_sIntersectionItem, m_cOcclusionCheckRay)) {
249  /* corner is occluded */
250  if(m_bShowTagRays) {
251  m_pcControllableEntity->GetCheckedRays().emplace_back(true, m_cOcclusionCheckRay);
252  }
253  /* no more checks necessary, move on to the next tag */
254  return true;
255  }
256  else {
257  /* corner not occluded */
258  if(m_bShowTagRays) {
259  m_pcControllableEntity->GetCheckedRays().emplace_back(false, m_cOcclusionCheckRay);
260  }
261  }
262  }
263  std::transform(std::begin(m_arrTagCorners),
264  std::end(m_arrTagCorners),
265  std::begin(m_arrTagCornerPixels),
266  [this] (const CVector3& c_tag_corner) {
267  return ProjectOntoSensor(c_tag_corner);
268  });
269  const CVector2& cCenterPixel = ProjectOntoSensor(c_tag.GetPosition());
270  /* try to convert tag payload to an unsigned integer */
271  UInt32 unId = 0;
272  try {
273  unId = std::stoul(c_tag.GetPayload());
274  }
275  catch(const std::logic_error& err_logic) {}
276  /* TODO: Update the position and orientation calculations to avoid the use of matrices */
277  CVector3 cTagPosition = m_cCameraToWorldTransform * c_tag.GetPosition();
278  CQuaternion cTagOrientation = m_cCameraOrientation.Inverse() * c_tag.GetOrientation();
279  /* transfer readings to the control interface */
280  m_vecTags.emplace_back(unId, cTagPosition, cTagOrientation, cCenterPixel, m_arrTagCornerPixels);
281  return true;
282  }

287  bool CPiPuckFrontCameraDefaultSensor::operator()(CDirectionalLEDEntity& c_led) {
288  if(c_led.GetColor() == CColor::BLACK) {
289  return true;

292  return true;
293  }

301  if(m_bShowLEDRays) {

316  CPiPuckFrontCameraDefaultSensor::ELedState
317  CPiPuckFrontCameraDefaultSensor::DetectLed(const CVector3& c_position) const {
318  /* c_position is the led in camera's coordinate system,
319  transfer it to global coordinate system */
320  CVector3 cLedPosition(c_position);
321  cLedPosition.Rotate(m_cCameraOrientation);
322  cLedPosition += m_cCameraPosition;
323  /* find the closest LED */
324  std::vector<SLed>::const_iterator itClosestLed =
325  std::min_element(std::cbegin(m_vecLedCache),
326  std::cend(m_vecLedCache),
327  [&cLedPosition] (const SLed& s_lhs_led, const SLed& s_rhs_led) {
328  return (Distance(s_lhs_led.Position, cLedPosition) <
329  Distance(s_rhs_led.Position, cLedPosition));
330  });
331  /* if no LEDs were found or if the closest LED is more than 0.5 cm away,
332  return ELedState::OFF */
333  if(itClosestLed == std::end(m_vecLedCache) ||
334  Distance(itClosestLed->Position, cLedPosition) > DETECT_LED_DIST_THRES) {
335  return ELedState::OFF;
336  }
337  /* At this point, we have the closest LED, estimate its state (mapped from
338  an exact color in simulation */
339  if(itClosestLed->Color == CColor::MAGENTA) {
340  return ELedState::Q1;
341  }
342  else if(itClosestLed->Color == CColor::ORANGE) {
343  return ELedState::Q2;
344  }
345  else if(itClosestLed->Color == CColor::GREEN) {
346  return ELedState::Q3;
347  }
348  else if(itClosestLed->Color == CColor::BLUE) {
349  return ELedState::Q4;
350  }
351  else {
352  return ELedState::OFF;
353  }
354  }

370  CVector2 CPiPuckFrontCameraDefaultSensor::ProjectOntoSensor(const CVector3& c_point) const {
371  CVector3 cCameraToEntityTranslation(m_cCameraToWorldTransform * c_point);
372  /* this could be avoided if CVector3 inherited from CMatrix<3,1> */
373  CMatrix<3,1> cCameraToEntityTranslationMatrix;
374  cCameraToEntityTranslationMatrix(0,0) =
375  cCameraToEntityTranslation.GetX() / cCameraToEntityTranslation.GetZ();
376  cCameraToEntityTranslationMatrix(1,0) =
377  cCameraToEntityTranslation.GetY() / cCameraToEntityTranslation.GetZ();
378  cCameraToEntityTranslationMatrix(2,0) = 1.0f;
379  /* get image coordinates */
380  CMatrix<3,1> cImageCoordinates(m_cProjectionMatrix * cCameraToEntityTranslationMatrix);
381  /* return as vector2 */
382  return CVector2(cImageCoordinates(0,0), cImageCoordinates(1,0));
