68  void CBatteryDefaultSensor::Update() {
69  /* sensor is disabled--nothing to do */
70  if (IsDisabled()) {
71  return;
72  }
73  /* Save old charge value (used later for time left estimation) */
74  Real fOldCharge = m_sReading.AvailableCharge;
75  /* Update available charge as seen by the robot */
76  m_sReading.AvailableCharge =
77  m_pcBatteryEntity->GetAvailableCharge() /
78  m_pcBatteryEntity->GetFullCharge();
79  /* Add noise */
80  if(m_bAddNoise) {
81  m_sReading.AvailableCharge += m_pcRNG->Uniform(m_cNoiseRange);
82  /* To trunc battery level between 0 and 1 */
83  UNIT.TruncValue(m_sReading.AvailableCharge);
84  }
85  /* Update time left */
86  if(m_sReading.AvailableCharge > 0.0) {
87  Real fDiff = fOldCharge - m_sReading.AvailableCharge;
88  if(Abs(fDiff) > 1e-6) {
89  m_sReading.TimeLeft =
90  fOldCharge *
91  CPhysicsEngine::GetSimulationClockTick() /
92  fDiff;
93  }
94  else {
95  m_sReading.TimeLeft = std::numeric_limits<Real>::infinity();
96  }
97  }
98  else {
99  m_sReading.TimeLeft = 0.0;
100  }
101  }
